cmake_minimum_required(VERSION 3.24)

project(learnopenglqt
DESCRIPTION "Step for step learning to use OpenGL with Qt"
LANGUAGES CXX)

# Lesson 1 will make a simple square move around the viewport
# a - using just the OpenGL wrapper
# b - using more OpenGL helper classes from Qt
add_subdirectory(Lesson_1/Lesson_1a)
add_subdirectory(Lesson_1/Lesson_1b)

# Lesson 2 will make a simple image based texture move around the viewport
# a - using just the OpenGL wrapper
# b - using more OpenGL helper classes from Qt
add_subdirectory(Lesson_2/Lesson_2a)
add_subdirectory(Lesson_2/Lesson_2b)

# Lesson 3 will make a simple 3D cube using texture move around 3D global space
# a - using just the OpenGL wrapper
# b - using more OpenGL helper classes from Qt
add_subdirectory(Lesson_3_3D/Lesson_3a)
add_subdirectory(Lesson_3_3D/Lesson_3b)
